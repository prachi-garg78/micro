.model small
.386
.data

ARRAY DB 10 DUP (?)
NUMBER DB (?)
LOW1 DW (?)
HIGH1 DW (?)
DATA DB (?)

MSG1 DB 'ENTER THE NO. OF ELEMENTS YOU WANT (0-9): $'
MSG2 DB 10,13,'ENTER ELEMENT: $'
MSG3 DB 10,13,'ENTER THE ELEMENT YOU WANT TO SEARCH: $'
MSG4 DB 10,13,'NUMBER IS FOUND AT POSITION: $'
MSG5 DB 10,13,'NUMBER NOT FOUND$'

.CODE
.STARTUP

MOV DX,OFFSET MSG1
MOV AH,09H
INT 21H

MOV EDX,0H
MOV EAX,0H
MOV SI,0

;TAKE THE NO. OF ELEMENTS
L1:
MOV AH,01H
INT 21H
SUB AL,30H
MOV NUMBER,AL
MOV AH,0
MOV CX,AX

MOV SI,0H

;ENTER THE ELEMENTS
L2:
MOV DX,OFFSET MSG2
MOV AH,09H
INT 21H
MOV BL,0H
MOV AH,01H
INT 21H
SUB AL,30H
CMP AL,09H
JBE L
SUB AL,07H
L:
MOV [SI],AL
INC SI
LOOP L2

MOV DX,OFFSET MSG3   ;PRINT MSG3
MOV AH,09H
INT 21H

;TAKES THE ELEMENT TO SEARCH
MOV BL,0
MOV AH,01H
INT 21H
SUB AL,30H
CMP AL,09H
JBE N
SUB AL,07H
N:
MOV DATA,AL
MOVZX CX,NUMBER

MOV DX,0H
MOV DX,DS
MOV ES,DX

MOV BP,CX
DEC BP
MOV LOW1,0H
MOV HIGH1,BP

L3:
MOV DX,HIGH1

CMP LOW1,DX
JA NF

ADD DX,LOW1
SHR DX,1
MOV BX,0H
MOV SI,DX
MOV BL,DATA
CMP BL,[SI]
JA ABOVE
JB BELOW
JE FOUND

ABOVE:
INC SI
MOV LOW1,SI
JMP L3

BELOW:
DEC SI
MOV HIGH1,SI
JMP L3

FOUND:
INC SI
MOV DX,OFFSET MSG4
MOV AH,09H
INT 21H

MOV DX,SI
CMP DX,09H
JBE NEXT
ADD DX,07H
NEXT:
ADD DX,30H
MOV AH,02H
INT 21H
JMP FINISH

NF:
MOV DX,OFFSET MSG5
MOV AH,09H
INT 21H

FINISH:

.EXIT
END
