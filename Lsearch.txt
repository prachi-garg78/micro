.model small
.386
.data

ARRAY DB 10 DUP (?)
NUMBER DB (?)
DATA DB (?)

MSG1 DB 10,13,'ENTER THE NO. OF ELEMENTS IN THE LIST (0-9): $'
MSG2 DB 10,13,'ENTER THE ELEMENT: $'
MSG3 DB 10,13,'ENTER THE ELEMENT YOU WANT TO SEARCH: $'
MSG4 DB 10,13,'NUMBER IS FOUND AT POSITION: $'
MSG5 DB 10,13,'NUMBER NOT FOUND$'


.CODE
.STARTUP

;print msg1
MOV DX,OFFSET MSG1
MOV AH,09H
INT 21H

MOV EDX,0H
MOV EAX,0H
MOV SI,0

;TAKE THE NO. OF ELEMENTS
MOV AH,01H
INT 21H
SUB AL,30H
MOV NUMBER,AL
MOV AH,0
MOV CX,AX

;print msg2
L2:
MOV DX,OFFSET MSG2
MOV AH,09H
INT 21H

MOV AH,01H
INT 21H
CMP AL,09H
JBE L
SUB AL,07H
L: 
SUB AL,30H
MOV ARRAY[SI],AL
INC SI
LOOP L2

;print msg3
MOV DX,OFFSET MSG3
MOV AH,09H
INT 21H

;TAKES THE ELEMENT TO SEARCH
MOV BL,0
MOV AH,01H
INT 21H
CMP AL,09H
JBE NEXT
SUB AL,07H
NEXT: 
SUB AL,30H

MOV DATA,AL
MOV BX,DS
MOV ES,BX
MOVZX AX,DATA

;SEARCHING THE ELEMENT
CLD
MOV DI,0
MOV DI,OFFSET ARRAY
MOVZX CX,NUMBER
INC CX
REPNE SCASB
CMP CX,0
JNE L4
MOV DX,OFFSET MSG5  ;ELEMENT NOT FOUND
MOV AH,09H
INT 21H
JMP L5
L4:
MOV DX,OFFSET MSG4   ;ELEMENT IS FOUND
MOV AH,09H
INT 21H

;PRINT POSITION
MOVZX DX,NUMBER
SUB DX,CX
INC DX
CMP DX,09
JBE N
ADD DX,07
N:
ADD DX,30H
MOV AH,02H
INT 21H

L5:
.EXIT
END
